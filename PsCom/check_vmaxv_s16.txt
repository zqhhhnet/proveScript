start:
    MOV R0, #A
    VMOV Q0, #B
    VMAXV.S16 R0, Q0
end

preCond:
    -2^5 <= A < 2^8
    0 <= B < 2^128
    -2^6 <= B00 < 2^12
    -25 <= B01 < 2515
    26 <= B02 < 232
    -24 <= B03 < 20
    -12 <= B04 < 52
    -11 <= B05 < 65
    -32 <= B06 < 156
    12 <= B07 < 601

postCond:
    0 <= R0 < 2^31
    0 <= Q0 < 2^128
